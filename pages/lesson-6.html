Async/Await

Синтаксис async/await для асинхронного программирования

Что такое Async/Await?

Async/Await - это синтаксический сахар над Promise, который позволяет писать асинхронный код как синхронный.

Основной синтаксис

// async функци�� всегда возвращает Promise
async function fetchData() {
    return 'data'; // Автоматически обёрнуто в Promise
}

fetchData().then(data => console.log(data)); // 'data'

await - ожидание выполнения

Ключевое слово `await` приостанавливает выполнение функции до разрешения Promise.

async function getUserData() {
    const response = await fetch('/api/user');
    const data = await response.json();
    console.log(data); // Выполнится после получения данных
    return data;
}

getUserData()
    .then(data => console.log('Got user:', data));

Обработка ошибок

try/catch

async function fetchUserSafe() {
    try {
        const response = await fetch('/api/user');
        if (!response.ok) {
            throw new Error(`HTTP error! status: ${response.status}`);
        }
        const data = await response.json();
        return data;
    } catch (error) {
        console.error('Failed to fetch user:', error);
        throw error; // Можно пробросить дальше
    }
}

await fetchUserSafe();

finally блок

async function fetchWithCleanup() {
    try {
        const response = await fetch('/data');
        return await response.json();
    } catch (error) {
        console.error(error);
    } finally {
        console.log('Request completed');
    }
}

Параллельное выполнение

Неправильно - последовательное

async function getSequential() {
    const user = await fetch('/user').then(r => r.json());      // 1 сек
    const posts = await fetch('/posts').then(r => r.json());    // 1 сек
    // Всего: 2 секунды
}

Правильно - параллельное

async function getParallel() {
    const [user, posts] = await Promise.all([
        fetch('/user').then(r => r.json()),
        fetch('/posts').then(r => r.json())
    ]);
    // Всего: 1 секунда
}

// Или с Promise.race для первого результата
async function getRaceResult() {
    const result = await Promise.race([
        fetch('/fast-api').then(r => r.json()),
        fetch('/slow-api').then(r => r.json())
    ]);
}

Async в разных контекстах

Методы класса

class UserService {
    async getUser(id) {
        const response = await fetch(`/api/users/${id}`);
        return response.json();
    }
    
    async getMultipleUsers(ids) {
        return Promise.all(ids.map(id => this.getUser(id)));
    }
}

const service = new UserService();
await service.getUser(1);

Стрелочные функции

const fetchData = async () => {
    const response = await fetch('/data');
    return response.json();
};

const numbers = [1, 2, 3];
const results = await Promise.all(
    numbers.map(async (num) => {
        const data = await fetch(`/api/${num}`);
        return data.json();
    })
);

Обработка нескольких ошибок

async function complexOperation() {
    try {
        const [user, settings] = await Promise.all([
            fetch('/user').then(r => r.json()).catch(e => {
                console.error('Failed to fetch user:', e);
                return null;
            }),
            fetch('/settings').then(r => r.json()).catch(e => {
                console.error('Failed to fetch settings:', e);
                return {};
            })
        ]);
        
        return { user, settings };
    } catch (error) {
        console.error('Operation failed:', error);
    }
}

Практические примеры

Загрузка данных с timeout

function timeout(ms) {
    return new Promise((_, reject) => 
        setTimeout(() => reject(new Error('Timeout')), ms)
    );
}

async function fetchWithTimeout(url, ms) {
    try {
        return await Promise.race([
            fetch(url).then(r => r.json()),
            timeout(ms)
        ]);
    } catch (error) {
        if (error.message === 'Timeout') {
            console.error(`Request timeout after ${ms}ms`);
        }
        throw error;
    }
}

await fetchWithTimeout('/data', 5000);

Retry с exponential backoff

async function retryWithBackoff(fn, maxRetries = 3, initialDelay = 1000) {
    for (let i = 0; i < maxRetries; i++) {
        try {
            return await fn();
        } catch (error) {
            if (i === maxRetries - 1) throw error;
            const delay = initialDelay * Math.pow(2, i);
            await new Promise(resolve => setTimeout(resolve, delay));
        }
    }
}

await retryWithBackoff(async () => fetch('/data').then(r => r.json()));
