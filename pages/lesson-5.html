Promises (Промисы)

Работа с промисами и обработка асинхронных операций

Что такое Promise?

Promise - это объект, который представляет результат асинхронной операции, которая может быть выполнена (resolved) или отклонена (rejected).

Три состояния Promise

// 1. Pending (ожидание) - начальное состояние
const pending = new Promise((resolve, reject) => {
    // операция выполняется
});

// 2. Resolved (выполнено) - успешно завершено
const resolved = new Promise((resolve, reject) => {
    resolve('Success!');
});

// 3. Rejected (отклонено) - завершено с ошибкой
const rejected = new Promise((resolve, reject) => {
    reject(new Error('Failed!'));
});

Создание Promise

const myPromise = new Promise((resolve, reject) => {
    const success = true;
    
    if (success) {
        resolve('Operation succeeded!');
    } else {
        reject(new Error('Operation failed!'));
    }
});

// Promise можно использовать только один раз
// resolve может быть вызвана либо reject, но не обе одновременно

then(), catch(), finally()

then() - обработка успешного результата

Promise.resolve('Hello')
    .then((result) => {
        console.log(result); // 'Hello'
        return result + ' World';
    })
    .then((result) => {
        console.log(result); // 'Hello World'
    });

catch() - обработка ошибок

Promise.reject(new Error('Oops!'))
    .catch((error) => {
        console.log(error.message); // 'Oops!'
        return 'Error handled';
    })
    .then((result) => {
        console.log(result); // 'Error handled'
    });

finally() - выполнение в любом случае

fetch('/data')
    .then((response) => response.json())
    .catch((error) => console.log(error))
    .finally(() => {
        console.log('Request completed (success or failure)');
    });

Цепочки Promise

function fetchUser(id) {
    return new Promise((resolve) => {
        setTimeout(() => resolve({ id, name: 'Alice' }), 1000);
    });
}

function fetchPosts(userId) {
    return new Promise((resolve) => {
        setTimeout(() => resolve([{ id: 1, title: 'Post 1' }]), 1000);
    });
}

fetchUser(1)
    .then((user) => {
        console.log('User:', user);
        return fetchPosts(user.id);
    })
    .then((posts) => {
        console.log('Posts:', posts);
    })
    .catch((error) => {
        console.error('Error:', error);
    });

Promise.all()

Ждёт выполнения всех промисов. Если хотя бы один отклонен, всё отклоняется.

const promise1 = Promise.resolve(3);
const promise2 = new Promise(resolve => setTimeout(() => resolve('foo'), 100));
const promise3 = fetch('/data').then(r => r.json());

Promise.all([promise1, promise2, promise3])
    .then((results) => {
        console.log(results); // [3, 'foo', { data }]
    })
    .catch((error) => {
        console.log('One of the promises rejected');
    });

Promise.allSettled()

Ждёт выполнения всех промисов, независимо от результата (resolved или rejected).

const promises = [
    Promise.resolve(1),
    Promise.reject('Error'),
    Promise.resolve(3)
];

Promise.allSettled(promises)
    .then((results) => {
        console.log(results);
        // [
        //   { status: 'fulfilled', value: 1 },
        //   { status: 'rejected', reason: 'Error' },
        //   { status: 'fulfilled', value: 3 }
        // ]
    });

Promise.race()

Возвращает результат первого завершённого промиса.

const promise1 = new Promise((resolve) => setTimeout(() => resolve('First'), 100));
const promise2 = new Promise((resolve) => setTimeout(() => resolve('Second'), 50));

Promise.race([promise1, promise2])
    .then((result) => {
        console.log(result); // 'Second' (быстрее завершился)
    });

Promise.any()

Возвращает первый успешно выполненный промис. Отклоняется только если все отклонены.

const promises = [
    Promise.reject('Error 1'),
    Promise.reject('Error 2'),
    Promise.resolve('Success')
];

Promise.any(promises)
    .then((result) => {
        console.log(result); // 'Success'
    })
    .catch((error) => {
        // AggregateError: All promises were rejected
    });

Практические примеры

Загрузка данных с retry

function fetchWithRetry(url, maxRetries = 3) {
    return fetch(url)
        .catch((error) => {
            if (maxRetries > 1) {
                return fetchWithRetry(url, maxRetries - 1);
            }
            throw error;
        });
}

fetchWithRetry('/data')
    .then(r => r.json())
    .then(data => console.log(data))
    .catch(error => console.error('Failed after retries:', error));

Упражнения

Упражнение 2: Напишите функцию `timeout(promise, ms)` которая отклоняет промис если он не выполнен за время.
Упражнение 3: Реализуйте функцию `retry(fn, times)` которая повторяет функцию при ошибке.
