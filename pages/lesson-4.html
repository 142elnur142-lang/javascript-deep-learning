Event Loop (Цикл событий)

Как работает event loop в JavaScript

Что такое Event Loop?

Event Loop - это механизм, который позволяет JavaScript выполнять асинхронный код несмотря на то, что язык является однопоточным.

Однопоточная природа JavaScript

console.log('Start');

setTimeout(() => {
    console.log('Timeout');
}, 0);

console.log('End');

// Вывод:
// Start
// End
// Timeout

Даже хотя setTimeout установлен на 0, он выполняется ПОСЛЕ синхронного кода. Почему?

Компоненты Event Loop

1. Call Stack (Стек вызовов)

Содержит код, который в данный момент выполняется.

function a() {
    b();
}

function b() {
    c();
}

function c() {
    console.log('c');
}

a();
// Стек: a() → b() → c() → console.log → (пусто)

2. Web APIs

Браузер предоставляет API (setTimeout, fetch, etc.), которые выполняются отдельно от JavaScript.

3. Callback Queue (Очередь обратных вызовов)

Содержит функции, готовые к выполнению.

4. Microtask Queue (Очередь микрозадач)

Приоритетнее Callback Queue. Содержит: Promise callbacks, MutationObserver, и т.д.

Порядок выполнения

console.log('1. Start');

Promise.resolve()
    .then(() => console.log('2. Promise (Microtask)'));

setTimeout(() => {
    console.log('3. setTimeout (Macrotask)');
}, 0);

console.log('4. End');

// Вывод:
// 1. Start
// 4. End
// 2. Promise (Microtask)
// 3. setTimeout (Macrotask)
⚠️ Важно: Microtasks выполняются ДО Macrotasks!

Микротаски vs Макротаски

Микротаски (Microtasks)

// Microtasks:
// - Promise.then/catch/finally
// - MutationObserver
// - queueMicrotask()
// - process.nextTick() (Node.js)

Promise.resolve()
    .then(() => console.log('Microtask 1'))
    .then(() => console.log('Microtask 2'));

Макротаски (Macrotasks)

// Macrotasks:
// - setTimeout
// - setInterval
// - setImmediate (Node.js)
// - requestAnimationFrame
// - I/O операции

setTimeout(() => console.log('Macrotask 1'), 0);
setTimeout(() => console.log('Macrotask 2'), 0);

Полный пример с объяснением

console.log('1');

setTimeout(() => {
    console.log('2'); // Macrotask
}, 0);

Promise.resolve()
    .then(() => {
        console.log('3'); // Microtask
        setTimeout(() => {
            console.log('4'); // Macrotask
        }, 0);
    })
    .then(() => {
        console.log('5'); // Microtask
    });

console.log('6');

// Вывод:
// 1 (синхронный)
// 6 (синхронный)
// 3 (microtask 1)
// 5 (microtask 2)
// 2 (macrotask 1)
// 4 (macrotask 2)

Визуализация Event Loop

// Event Loop цикл:
// 1. Выполнить все синхронный код (Call Stack)
// 2. Проверить Microtask Queue - выполнить все
// 3. Проверить Macrotask Queue - выполнить одну
// 4. Вернуться на шаг 2

// Это означает:
console.log('A');

setTimeout(() => {
    console.log('B'); // Первый macrotask
    Promise.resolve().then(() => {
        console.log('C'); // Microtask (выполнится ДО следующего macrotask)
    });
}, 0);

setTimeout(() => {
    console.log('D'); // Второй macrotask
}, 0);

console.log('E');

// Вывод: A, E, B, C, D

Практические применения

1. Гарантированный асинхронный код

// Гарантирует выполнение в следующем цикле event loop
function nextTick(callback) {
    return Promise.resolve().then(callback);
}

console.log('A');
nextTick(() => console.log('B'));
console.log('C');

// Вывод: A, C, B

2. Оптимизация с requestAnimationFrame

// requestAnimationFrame выполняется перед рендерингом
requestAnimationFrame(() => {
    console.log('Paint');
});

console.log('Script');

// Вывод: Script, затем Paint (перед рендерингом)
