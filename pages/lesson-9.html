Scope Chain - Цепочка областей

Как JavaScript ищет переменные в разных областях видимости

Что такое Scope?

Scope определяет где в коде доступна переменная.

const global = 'global';

function outer() {
    const outer = 'outer';
    
    function inner() {
        const inner = 'inner';
        console.log(inner, outer, global);
    }
    
    inner();
}

outer();
