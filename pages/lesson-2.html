Prototypes (Прототипы)

Прототипное наследование и цепочка прототипов

Что такое прототип?

Прототип - это объект, который служит источником для свойств и методов других объектов. В JavaScript каждый объект имеет скрытое свойство [[Prototype]].

Основной пример

const animal = {
    speak: function() {
        console.log('Animal speaks');
    }
};

const dog = Object.create(animal);
dog.name = 'Rex';

dog.speak(); // 'Animal speaks'
console.log(dog.name); // 'Rex'

Цепочка прототипов (Prototype Chain)

JavaScript ищет свойство сначала в самом объекте, затем в его прототипе, затем в прототипе прототипа и так далее.

const parent = { a: 1 };
const child = Object.create(parent);
child.b = 2;
const grandchild = Object.create(child);
grandchild.c = 3;

console.log(grandchild.a); // 1 (из parent)
console.log(grandchild.b); // 2 (из child)
console.log(grandchild.c); // 3 (собственное)

Constructor Functions

Традиционный способ создания объектов с прототипом - использование конструктор-функций.

Пример конструктор-функции

function Person(name, age) {
    this.name = name;
    this.age = age;
}

Person.prototype.greet = function() {
    console.log(`Hi, I'm ${this.name}`);
};

const person1 = new Person('Alice', 30);
const person2 = new Person('Bob', 25);

person1.greet(); // 'Hi, I'm Alice'
person2.greet(); // 'Hi, I'm Bob'

Object.create vs Constructor Functions

Использование Object.create

const animalProto = {
    speak() {
        console.log(this.name + ' speaks');
    }
};

function createDog(name) {
    const dog = Object.create(animalProto);
    dog.name = name;
    return dog;
}

const myDog = createDog('Rex');
myDog.speak(); // 'Rex speaks'

Проверка прототипов

function Animal() {}
const dog = new Animal();

// Проверка прототипа
console.log(dog instanceof Animal); // true
console.log(Object.getPrototypeOf(dog) === Animal.prototype); // true
console.log(dog.constructor === Animal); // true

// Проверка собственных свойств
console.log(dog.hasOwnProperty('name')); // false

Наследование через прототипы

function Animal(name) {
    this.name = name;
}

Animal.prototype.speak = function() {
    console.log(this.name + ' makes a sound');
};

function Dog(name, breed) {
    Animal.call(this, name); // Вызов родительского конструктора
    this.breed = breed;
}

// Наследование
Dog.prototype = Object.create(Animal.prototype);
Dog.prototype.constructor = Dog;

Dog.prototype.bark = function() {
    console.log(this.name + ' barks');
};

const myDog = new Dog('Rex', 'Labrador');
myDog.speak(); // 'Rex makes a sound'
myDog.bark();  // 'Rex barks'
⚠️ Важно: При переопределении прототипа всегда восстанавливайте свойство `constructor`.

Практические примеры

Полнофункциональный класс с наследованием

function Vehicle(type) {
    this.type = type;
}

Vehicle.prototype.getType = function() {
    return this.type;
};

function Car(type, doors) {
    Vehicle.call(this, type);
    this.doors = doors;
}

Car.prototype = Object.create(Vehicle.prototype);
Car.prototype.constructor = Car;

Car.prototype.getInfo = function() {
    return `${this.getType()} with ${this.doors} doors`;
};

const myCar = new Car('Sedan', 4);
console.log(myCar.getInfo()); // 'Sedan with 4 doors'
